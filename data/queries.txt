LOAD EMPLOYEE
EMPLOYEE_HIGH_SALARY <- SELECT Salary >= 30000 FROM EMPLOYEE 
Q1 <- PROJECT Ssn, Salary FROM EMPLOYEE_HIGH_SALARY




EMPLOYEE_CROSS_EMPLOYEE <- CROSS EMPLOYEE EMPLOYEE
EMPLOYEE_CROSS_EMPLOYEE_SSN <- SELECT EMPLOYEE1_Super_ssn == EMPLOYEE2_Ssn FROM EMPLOYEE_CROSS_EMPLOYEE 
EMPLOYEE_CROSS_EMPLOYEE_SSN_DATE <- SELECT EMPLOYEE1_Bdate == EMPLOYEE2_Bdate FROM EMPLOYEE_CROSS_EMPLOYEE_SSN
Q2 <- PROJECT EMPLOYEE1_Ssn, EMPLOYEE1_Super_ssn, EMPLOYEE1_Bdate FROM EMPLOYEE_CROSS_EMPLOYEE_SSN_DATE
RENAME EMPLOYEE1_Ssn TO Ssn FROM Q2
RENAME EMPLOYEE1_Super_ssn TO Super_ssn FROM Q2
RENAME EMPLOYEE1_Bdate TO Bdate FROM Q2




LOAD WORKS_ON
EMPLOYEE_CROSS_EMPLOYEE_SEX <- SELECT EMPLOYEE1_Sex != EMPLOYEE2_Sex FROM EMPLOYEE_CROSS_EMPLOYEE 
WORKS_CROSS_WORKS <- CROSS WORKS_ON WORKS_ON
WORKS_CROSS_WORKS-PROJECT <- SELECT WORKS_ON1_Pno == WORKS_ON2_Pno FROM WORKS_CROSS_WORKS


EE_CROSS_WW <- CROSS EMPLOYEE_CROSS_EMPLOYEE_SEX WORKS_CROSS_WORKS-PROJECT
TEMP_Q3_1 <- SELECT EMPLOYEE1_Ssn == WORKS_ON1_Essn FROM EE_CROSS_WW 
TEMP_Q3_2 <- SELECT EMPLOYEE2_Ssn == WORKS_ON2_Essn FROM TEMP_Q3_1
TEMP_Q3_3 <- SELECT EMPLOYEE1_Ssn != EMPLOYEE2_Ssn FROM TEMP_Q3_2

Q3_1 <- PROJECT EMPLOYEE1_Ssn, EMPLOYEE2_Ssn, WORKS_ON2_Pno FROM TEMP_Q3_3
Q3 <- PROJECT WORKS_ON2_Pno, FROM Q3_1
RENAME WORKS_ON2_Pno TO Pno FROM Q3




EE_CROSS_WW_SSN <- CROSS EMPLOYEE_CROSS_EMPLOYEE_SSN WORKS_CROSS_WORKS-PROJECT
TEMP_Q4_1 <- SELECT EMPLOYEE1_Ssn == WORKS_ON1_Essn FROM EE_CROSS_WW_SSN 
TEMP_Q4_2 <- SELECT EMPLOYEE2_Ssn == WORKS_ON2_Essn FROM TEMP_Q4_1
Q4 <- PROJECT EMPLOYEE1_Ssn, EMPLOYEE1_Super_ssn, WORKS_ON2_Pno FROM TEMP_Q4_2
RENAME EMPLOYEE1_Ssn TO Ssn FROM Q4
RENAME EMPLOYEE1_Super_ssn TO Super_ssn FROM Q4
RENAME WORKS_ON2_Pno TO Pno FROM Q4




LOAD PROJECT
Q5 <- SELECT Dnum != 4 FROM PROJECT
